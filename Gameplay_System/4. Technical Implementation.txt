4. Technical Implementation
===========================

4.1 Game Engine
---------------
The game will be developed using Unity, selected for its robust support for both 2D and 3D game development, extensive community resources, and proven success in similar titles like *Kerbal Space Program* and *Rimworld*. Unity's flexibility enables seamless integration of the game's dual mechanics: 3D orbital navigation and 2D crew management.

- **Key Features:**
  - 3D rendering for the solar system simulation.
  - 2D tools (sprite rendering, tilemaps) for the ship's interior.
  - Physics engine for realistic orbital mechanics (augmented with plugins).
  - Scripting in C# for gameplay logic and AI behavior.

- **Third-Party Plugins:**
  - **Simple Kepler Orbits:** For simulating realistic orbital mechanics.
  - **Yarn Spinner:** For managing dialogue trees and narrative branching.

4.2 Orbital Mechanics
---------------------
Orbital mechanics will be implemented using the Simple Kepler Orbits plugin from the Unity Asset Store, which provides tools for simulating spacecraft trajectories based on Keplerian mechanics.

- **Trajectory Plotting:**
  - Players plot courses using a 3D interface, selecting pre-calculated paths or adjusting parameters (e.g., delta-v, burn timing).
  - Crew members (e.g., Pilot) can suggest course corrections based on new data (e.g., rival ship movements).

- **Propulsion:**
  - Spacecraft use chemical rockets, with realistic fuel consumption and transit times (e.g., 6-9 months to Mars).

- **EVA System:**
  - An extravehicular activity (EVA) feature allows crew to perform tasks outside the ship (e.g., repairs, rescues).
  - Crew are represented as 3D models during EVA, with success rates tied to skills and traits (e.g., Engineering skill, Calm trait).

**4.2.1 Combat Trajectory System**
----------------------------------
The space combat system implements realistic orbital mechanics for weapons fire, target interception, and evasive maneuvers, ensuring that gameplay accurately reflects the constraints of physics in space.

- **Intercept Calculation Systems:**
  ```csharp
  /// <summary>
  /// Calculates an intercept solution for a target moving along a predictable orbital trajectory.
  /// </summary>
  /// <param name="firingPosition">The current position of the firing vessel</param>
  /// <param name="firingVelocity">The current velocity vector of the firing vessel</param>
  /// <param name="targetOrbit">The orbital parameters of the target vessel</param>
  /// <param name="projectileSpeed">Speed of the projectile in m/s</param>
  /// <param name="maxTimeToIntercept">Maximum time window to consider for intercept solutions</param>
  /// <returns>An InterceptSolution containing firing vector and time to intercept, null if no solution found</returns>
  public InterceptSolution CalculateOrbitalIntercept(
      Vector3 firingPosition, 
      Vector3 firingVelocity,
      KeplerOrbit targetOrbit,
      float projectileSpeed,
      float maxTimeToIntercept)
  {
      // Start with a coarse-grained search for possible intercept times
      List<float> potentialInterceptTimes = new List<float>();
      float timeStep = maxTimeToIntercept / INITIAL_SEARCH_STEPS;
      
      for (float t = 0; t <= maxTimeToIntercept; t += timeStep)
      {
          // Calculate target position at time t
          Vector3 targetPosition = targetOrbit.GetPositionAtTime(
              targetOrbit.TimeOfPeriapsis + t
          );
          
          // Calculate travel time for projectile to reach this position
          float distanceToTarget = Vector3.Distance(firingPosition, targetPosition);
          float travelTime = distanceToTarget / projectileSpeed;
          
          // If projectile would arrive close to time t, this is a potential solution
          if (Mathf.Abs(travelTime - t) < COARSE_SEARCH_THRESHOLD)
          {
              potentialInterceptTimes.Add(t);
          }
      }
      
      // Refine each potential solution using Newton-Raphson iteration
      InterceptSolution bestSolution = null;
      float lowestDeltaV = float.MaxValue;
      
      foreach (float initialGuess in potentialInterceptTimes)
      {
          InterceptSolution solution = RefineInterceptSolution(
              firingPosition,
              firingVelocity,
              targetOrbit,
              projectileSpeed,
              initialGuess
          );
          
          if (solution != null)
          {
              // Calculate required delta-V to achieve this firing solution
              float requiredDeltaV = Vector3.Magnitude(
                  solution.firingVector - firingVelocity
              );
              
              // Keep track of the solution requiring least delta-V
              if (requiredDeltaV < lowestDeltaV)
              {
                  lowestDeltaV = requiredDeltaV;
                  bestSolution = solution;
              }
          }
      }
      
      return bestSolution;
  }
  
  /// <summary>
  /// Refines an intercept solution using Newton-Raphson iteration.
  /// </summary>
  private InterceptSolution RefineInterceptSolution(
      Vector3 firingPosition,
      Vector3 firingVelocity,
      KeplerOrbit targetOrbit,
      float projectileSpeed,
      float initialTimeGuess)
  {
      float timeGuess = initialTimeGuess;
      float previousError = float.MaxValue;
      
      for (int i = 0; i < MAX_REFINEMENT_ITERATIONS; i++)
      {
          // Calculate target position at timeGuess
          Vector3 targetPosition = targetOrbit.GetPositionAtTime(
              targetOrbit.TimeOfPeriapsis + timeGuess
          );
          
          // Calculate distance and travel time
          float distanceToTarget = Vector3.Distance(firingPosition, targetPosition);
          float travelTime = distanceToTarget / projectileSpeed;
          
          // Calculate error (difference between arrival time and target time)
          float error = travelTime - timeGuess;
          
          // Check if we've converged to an acceptable solution
          if (Mathf.Abs(error) < CONVERGENCE_THRESHOLD)
          {
              // Calculate firing vector required to hit target
              Vector3 firingDirection = (targetPosition - firingPosition).normalized;
              Vector3 firingVector = firingDirection * projectileSpeed;
              
              return new InterceptSolution {
                  firingVector = firingVector,
                  timeToIntercept = travelTime,
                  targetPosition = targetPosition
              };
          }
          
          // If error is increasing, the iteration is diverging
          if (Mathf.Abs(error) > Mathf.Abs(previousError))
          {
              break;
          }
          
          // Update time guess using Newton-Raphson step
          // Derivative of error function is approximately -1 + derivative of distance/speed
          // For simplicity, we use a fixed derivative approximation
          float derivativeApprox = -1.2f;
          timeGuess = timeGuess - error / derivativeApprox;
          
          previousError = error;
      }
      
      // Failed to converge to a solution
      return null;
  }

- **Heat Signature Management:**
  The combat system includes heat generation, dissipation, and detection mechanics to model the thermal signatures of spacecraft during engagements.
  
  ```csharp
  /// <summary>
  /// Updates heat generation and dissipation for a combat vessel.
  /// </summary>
  public void UpdateThermalSignature(CombatVessel vessel, float deltaTime)
  {
      // Calculate heat generation from active systems
      float heatGeneration = 0f;
      
      // Main propulsion heat
      if (vessel.mainEngineActive)
      {
          heatGeneration += vessel.mainEngineThrustPercentage * 
                           vessel.heatGenerationCurve.Evaluate(vessel.mainEngineThrustPercentage) * 
                           vessel.mainEngineHeatOutput;
      }
      
      // RCS thrusters heat
      heatGeneration += vessel.rcsActivityLevel * vessel.rcsHeatOutput;
      
      // Weapon systems heat
      foreach (WeaponSystem weapon in vessel.weaponSystems)
      {
          if (weapon.isActive)
          {
              heatGeneration += weapon.currentHeatGeneration;
          }
      }
      
      // Life support and electronics baseline heat
      heatGeneration += vessel.baselineHeatGeneration;
      
      // Calculate heat dissipation
      // Radiators effectiveness varies based on orientation to the sun
      float radiatorEfficiency = CalculateRadiatorEfficiency(
          vessel.transform.rotation, 
          vessel.sunDirection
      );
      
      float heatDissipation = vessel.radiatorCapacity * 
                              radiatorEfficiency * 
                              vessel.radiatorDeploymentPercentage;
      
      // Calculate net heat change
      float netHeatChange = heatGeneration - heatDissipation;
      
      // Update vessel temperature
      vessel.currentHeat += netHeatChange * deltaTime;
      vessel.currentHeat = Mathf.Clamp(
          vessel.currentHeat, 
          AMBIENT_TEMPERATURE, 
          vessel.maximumSafeTemperature * 1.5f
      );
      
      // Update infrared signature based on temperature
      vessel.infraredSignature = CalculateInfraredSignature(
          vessel.currentHeat, 
          vessel.hullArea, 
          vessel.hullEmissivity
      );
      
      // Check for overheating damage
      if (vessel.currentHeat > vessel.maximumSafeTemperature)
      {
          float overHeatRatio = (vessel.currentHeat - vessel.maximumSafeTemperature) / 
                              (vessel.maximumSafeTemperature * 0.5f);
          
          ApplyOverheatDamage(vessel, overHeatRatio * OVERHEAT_DAMAGE_RATE * deltaTime);
      }
  }
  
  /// <summary>
  /// Calculates the detection probability of a vessel based on its thermal signature.
  /// </summary>
  public float CalculateDetectionProbability(
      CombatVessel target, 
      CombatVessel observer, 
      float distance)
  {
      // Base detection probability falls off with square of distance
      float baseProbability = 1.0f / (1.0f + (distance * distance / 
                                            (observer.sensorRange * observer.sensorRange)));
      
      // Adjust for target's thermal signature
      float signatureMultiplier = target.infraredSignature / STANDARD_THERMAL_SIGNATURE;
      
      // Adjust for sensor quality and crew skill
      float sensorEfficiency = observer.baseSensorEfficiency;
      CrewMember sensorOperator = observer.GetSystemOperator(ShipSystem.Sensors);
      
      if (sensorOperator != null)
      {
          sensorEfficiency *= 1.0f + (sensorOperator.GetSkill(SkillType.Sensors) / 100.0f);
      }
      
      // Calculate final detection probability
      float detectionProbability = baseProbability * signatureMultiplier * sensorEfficiency;
      
      // Environmental modifiers (solar radiation, dust fields, etc.)
      detectionProbability *= CalculateEnvironmentalDetectionModifier(
          observer.currentRegion, 
          observer.currentSolarActivity
      );
      
      return Mathf.Clamp01(detectionProbability);
  }

- **Debris Field Generation:**
  Combat events generate persistent orbital debris that affects navigation and future engagements in the same region.
  
  ```csharp
  /// <summary>
  /// Generates orbital debris from a ship-to-ship weapon impact or destruction event.
  /// </summary>
  /// <param name="position">Impact position in space</param>
  /// <param name="impactForce">Force of the impact in newtons</param>
  /// <param name="targetMass">Mass of the impacted vessel or component</param>
  /// <param name="referenceOrbit">The orbital parameters of the target at time of impact</param>
  public void GenerateCombatDebris(
      Vector3 position, 
      float impactForce, 
      float targetMass, 
      KeplerOrbit referenceOrbit)
  {
      // Calculate debris field properties
      int debrisCount = Mathf.FloorToInt(
          Mathf.Sqrt(impactForce * targetMass) / DEBRIS_GENERATION_FACTOR
      );
      
      debrisCount = Mathf.Clamp(debrisCount, MIN_DEBRIS_COUNT, MAX_DEBRIS_COUNT);
      
      // Create orbital debris objects
      for (int i = 0; i < debrisCount; i++)
      {
          // Calculate ejection velocity (direction and magnitude)
          Vector3 ejectionDirection = Random.onUnitSphere;
          float ejectionSpeed = Random.Range(
              MIN_EJECTION_SPEED,
              Mathf.Sqrt(impactForce / targetMass) * EJECTION_SPEED_FACTOR
          );
          
          Vector3 ejectionVelocity = ejectionDirection * ejectionSpeed;
          
          // Calculate debris size and mass
          float debrisSize = Random.Range(MIN_DEBRIS_SIZE, MAX_DEBRIS_SIZE);
          float debrisMass = debrisSize * debrisSize * DEBRIS_DENSITY;
          
          // Create new orbital parameters based on reference orbit plus ejection velocity
          Vector3 newVelocity = referenceOrbit.GetVelocityAtTime(
              referenceOrbit.GetTimeAtTrueAnomaly(referenceOrbit.TrueAnomaly)
          ) + ejectionVelocity;
          
          KeplerOrbit debrisOrbit = new KeplerOrbit(
              referenceOrbit.centralBody,
              position,
              newVelocity,
              Time.time
          );
          
          // Calculate debris lifetime based on orbit parameters
          float lifetime = CalculateDebrisLifetime(
              debrisOrbit,
              debrisSize,
              debrisMass
          );
          
          // Create and register the debris object
          OrbitalDebris debris = new OrbitalDebris(
              debrisOrbit,
              debrisSize,
              debrisMass,
              Time.time,
              Time.time + lifetime
          );
          
          worldState.debrisManager.RegisterDebris(debris);
          
          // Update regional hazard level for navigation
          SpaceRegion region = worldState.GetRegionAtPosition(position);
          region.hazardLevel += debrisSize * HAZARD_MULTIPLIER;
      }
      
      // Apply regional effects
      ApplyDebrisFieldEffects(position, debrisCount);
  }
  
  /// <summary>
  /// Calculates the probability of a collision with orbital debris during transit.
  /// </summary>
  public float CalculateDebrisCollisionProbability(
      KeplerOrbit shipOrbit,
      float transitTime,
      float shipCrossSectionalArea)
  {
      float collisionProbability = 0f;
      
      // Get all debris regions that intersect with the ship's orbit
      List<SpaceRegion> regionsAlongPath = worldState.GetRegionsAlongOrbit(shipOrbit);
      
      foreach (SpaceRegion region in regionsAlongPath)
      {
          // Calculate time spent in each region
          float timeInRegion = CalculateTimeInRegion(shipOrbit, region);
          
          // Calculate collision probability based on debris density in region
          float regionalProbability = region.debrisDensity * 
                                     shipCrossSectionalArea * 
                                     timeInRegion * 
                                     COLLISION_COEFFICIENT;
          
          // Account for ship velocity relative to debris field
          float velocityFactor = CalculateRelativeVelocityFactor(
              shipOrbit, 
              region.averageDebrisOrbit
          );
          
          regionalProbability *= velocityFactor;
          
          // Accumulate total probability
          collisionProbability += regionalProbability;
      }
      
      return Mathf.Clamp01(collisionProbability);
  }
  ```

- **Realistic Weapon Modeling:**
  Weapon systems are modeled with physics-based constraints that affect targeting, damage, and effectiveness at different ranges.
  
  ```csharp
  /// <summary>
  /// Models the behavior of a kinetic projectile in space.
  /// </summary>
  public void UpdateKineticProjectile(KineticProjectile projectile, float deltaTime)
  {
      // Update position based on velocity
      projectile.position += projectile.velocity * deltaTime;
      
      // Apply gravitational influences from nearby bodies
      foreach (CelestialBody body in bodiesInRange)
      {
          Vector3 bodyDirection = body.position - projectile.position;
          float distanceSquared = bodyDirection.sqrMagnitude;
          
          // Skip if too far for meaningful influence
          if (distanceSquared > body.gravityInfluenceRadiusSquared)
              continue;
              
          // Calculate gravitational acceleration
          float gravitationalAcceleration = GRAVITATIONAL_CONSTANT * 
                                          body.mass / 
                                          distanceSquared;
                                          
          Vector3 accelerationVector = bodyDirection.normalized * gravitationalAcceleration;
          
          // Apply acceleration to velocity
          projectile.velocity += accelerationVector * deltaTime;
      }
      
      // Check for target hit
      if (projectile.targetTracking && projectile.target != null)
      {
          float distanceToTarget = Vector3.Distance(
              projectile.position, 
              projectile.target.position
          );
          
          if (distanceToTarget < projectile.hitThreshold)
          {
              // Calculate impact damage based on kinetic energy
              float kineticEnergy = 0.5f * 
                                   projectile.mass * 
                                   projectile.velocity.sqrMagnitude;
                                   
              // Apply damage to target
              ApplyKineticDamage(
                  projectile.target,
                  kineticEnergy,
                  projectile.velocity.normalized,
                  projectile.armorPenetration
              );
              
              // Generate impact debris
              GenerateCombatDebris(
                  projectile.target.position,
                  kineticEnergy,
                  projectile.target.mass,
                  projectile.target.orbit
              );
              
              // Destroy projectile
              RemoveProjectile(projectile);
              return;
          }
      }
      
      // Update lifetime and check if expired
      projectile.lifetime -= deltaTime;
      if (projectile.lifetime <= 0f)
      {
          RemoveProjectile(projectile);
      }
  }
  
  /// <summary>
  /// Models the behavior of a directed energy weapon (laser) in space.
  /// </summary>
  public void FireDirectedEnergyWeapon(
      DirectedEnergyWeapon weapon, 
      CombatVessel target,
      float firingDuration)
  {
      // Calculate base damage
      float basePower = weapon.outputPower * firingDuration;
      
      // Calculate beam divergence based on distance
      float distance = Vector3.Distance(weapon.position, target.position);
      float beamRadius = weapon.initialBeamRadius + 
                       (distance * Mathf.Tan(weapon.divergenceAngle * Mathf.Deg2Rad));
      
      // Calculate power density at target
      float beamArea = Mathf.PI * beamRadius * beamRadius;
      float powerDensity = basePower / beamArea;
      
      // Account for atmospheric interference if applicable
      if (target.isInAtmosphere || weapon.isInAtmosphere)
      {
          powerDensity *= CalculateAtmosphericTransmission(
              weapon.position,
              target.position,
              weapon.wavelength
          );
      }
      
      // Calculate targeting accuracy based on crew skill and jitter
      float accuracy = CalculateBeamAccuracy(
          weapon,
          distance,
          weapon.GetOperator()?.GetSkill(SkillType.Weapons) ?? 0
      );
      
      // Apply accuracy modifier to power density
      powerDensity *= accuracy;
      
      // Calculate damage based on power density and target armor properties
      float damage = CalculateDirectedEnergyDamage(
          powerDensity,
          target.armorReflectivity,
          target.armorThermalConductivity,
          target.armorMeltingPoint,
          firingDuration
      );
      
      // Apply damage to target
      ApplyDirectedEnergyDamage(target, damage, weapon.position);
      
      // Generate heat in the weapon system
      weapon.currentHeat += weapon.heatGenerationPerSecond * firingDuration;
      
      // Update weapon cooldown
      weapon.cooldownTimeRemaining = weapon.baseCooldownTime;
  }

4.3 Crew Management System
--------------------------
Crew management will be handled in a 2D top-down view of the ship's interior, leveraging Unity's 2D tools.

- **2D Interior View:**
  - Side-facing and back-facing sprites for a cutaway effect, similar to *Barotrauma*.
  - Players can move via WASD or click interactable objects (e.g., computers, panels) for tasks like scanning or comms.

- **Zero-G Movement System:**
  - Crew movement in zero gravity follows realistic physics:
    * Point-and-click navigation where crew push off and float to destinations
    * Momentum-based movement with gradual deceleration using handholds
    * Trajectory visualization showing projected paths before movement commands
    * Collision avoidance with other crew members and obstacles
  
  - **Implementation Details:**
    ```csharp
    void MoveCrewMember(CrewMember crew, Vector2 targetPosition) {
        // Calculate launch vector (direction and power)
        Vector2 moveDirection = (targetPosition - crew.position).normalized;
        float distance = Vector2.Distance(crew.position, targetPosition);
        
        // Adjust push-off power based on distance and crew attributes
        float pushPower = Mathf.Clamp(
            distance * BASE_PUSH_COEFFICIENT,
            MIN_PUSH_POWER,
            crew.GetAttributeValue(Attribute.Strength) * MAX_PUSH_MULTIPLIER
        );
        
        // Apply impulse force in the movement direction
        crew.rigidbody.AddForce(moveDirection * pushPower, ForceMode2D.Impulse);
        
        // Calculate and visualize trajectory
        DrawTrajectory(crew, moveDirection * pushPower);
        
        // Set crew state to "Moving"
        crew.SetState(CrewState.Moving);
        
        // Schedule grab action on arrival near destination
        float estimatedArrivalTime = CalculateArrivalTime(
            distance, 
            pushPower, 
            crew.rigidbody.drag
        );
        
        // Schedule deceleration when approaching target
        StartCoroutine(BeginDecelerationAtTarget(
            crew, 
            targetPosition,
            estimatedArrivalTime * 0.8f
        ));
    }
    
    IEnumerator BeginDecelerationAtTarget(CrewMember crew, Vector2 target, float delay) {
        yield return new WaitForSeconds(delay);
        
        // Check if we're approaching target and still in motion
        if (crew.GetState() == CrewState.Moving && 
            Vector2.Distance(crew.position, target) < DECELERATION_DISTANCE) {
            
            // Find nearest handhold
            Handhold nearestGrab = FindNearestHandhold(crew.position, target);
            
            if (nearestGrab != null) {
                // Adjust crew trajectory to intercept handhold
                Vector2 adjustedDirection = (nearestGrab.position - crew.position).normalized;
                crew.rigidbody.velocity = adjustedDirection * 
                                        crew.rigidbody.velocity.magnitude * 
                                        COURSE_CORRECTION_FACTOR;
                
                // Trigger grab animation when in range
                StartCoroutine(GrabHandholdWhenInRange(crew, nearestGrab));
            } else {
                // No handhold available, activate RCS for stopping
                ActivateRCS(crew, target);
            }
        }
    }
    
    IEnumerator GrabHandholdWhenInRange(CrewMember crew, Handhold handhold) {
        // Wait until within grab range
        while (Vector2.Distance(crew.position, handhold.position) > GRAB_DISTANCE) {
            yield return null;
        }
        
        // Grab handhold
        crew.rigidbody.velocity = Vector2.zero;
        crew.transform.position = handhold.position;
        crew.SetState(CrewState.Idle);
        crew.PlayAnimation("Grab");
        
        // Apply stress/stamina effects
        float movementEffort = crew.initialMoveVelocity.magnitude * EFFORT_FACTOR;
        crew.ApplyStamina(-movementEffort);
        
        // Trigger onArrival events
        if (crew.onArrivalAction != null) {
            crew.onArrivalAction.Invoke();
        }
    }
    
    void ActivateRCS(CrewMember crew, Vector2 targetPosition) {
        // Calculate deceleration vector (opposite to current velocity)
        Vector2 decelerationDirection = -crew.rigidbody.velocity.normalized;
        
        // Apply RCS thrust (weaker than push-off, consumes RCS fuel)
        float rcsThrust = crew.rigidbody.velocity.magnitude * RCS_THRUST_FACTOR;
        crew.rigidbody.AddForce(decelerationDirection * rcsThrust, ForceMode2D.Force);
        
        // Consume RCS fuel
        crew.ConsumeRcsCharge(rcsThrust * RCS_FUEL_CONSUMPTION_RATE);
        
        // Play RCS effect
        CreateRcsThrustEffect(crew.position, decelerationDirection);
        
        // Set crew state to "Decelerating"
        crew.SetState(CrewState.Decelerating);
    }

  - **Physical Simulation:**
    * Realistic momentum conservation with drag coefficients based on environment (habitation/industrial/airlock zones)
    * Variable movement speed based on crew attributes (EVA skill modifies control precision: +/-15%)
    * Fatigue system where consecutive movements without rest increase error margin (5% per movement, max 25%)
    * Injury affects movement precision (broken limb: -30% navigation accuracy, +50% deceleration time)

- **Emergent Behavior System:**
  - Crew members exhibit dynamic behaviors derived from their traits, skills, and current state:
    * Need-based autonomy where crew satisfy personal requirements (rest, food, social interaction)
    * Trait-driven interactions with environment and other crew
    * Experience-based learning adjusting behavior patterns over time
    * Contextual awareness triggering appropriate responses to emergencies or opportunities
  
  - **Implementation Details:**

    void UpdateExperienceAndLearning(CrewMember crew, float deltaTime) {
        // Check if crew is performing a skill-based action
        if (crew.currentAction != null && crew.currentAction.providesExperience) {
            // Accumulate experience in the relevant skill
            SkillType relevantSkill = crew.currentAction.relatedSkill;
            crew.AddExperience(relevantSkill, EXPERIENCE_GAIN_RATE * deltaTime);
            
            // Chance to improve at task when meaningful experience threshold is reached
            if (crew.GetExperience(relevantSkill) >= SKILL_IMPROVEMENT_THRESHOLD) {
                // Improve skill and reset experience counter
                crew.ImproveSkill(relevantSkill, SKILL_IMPROVEMENT_AMOUNT);
                crew.ResetExperience(relevantSkill);
                
                // Notify player of skill improvement
                notificationSystem.CreateSkillImprovedNotification(crew, relevantSkill);
            }
        }
        
        // Learning from observation (watching others perform tasks)
        if (crew.GetState() == CrewState.Observing && crew.observingAction != null) {
            CrewMember observedCrew = crew.observingCrewMember;
            SkillType observedSkill = crew.observingAction.relatedSkill;
            
            // Gain experience at a reduced rate when observing
            float observationExperienceRate = OBSERVATION_EXPERIENCE_FACTOR * 
                                           EXPERIENCE_GAIN_RATE * 
                                           deltaTime;
            
            crew.AddExperience(observedSkill, observationExperienceRate);
        }
    }

            .setEase(LeanTweenType.easeInOutSine);
        
        yield return new WaitForSeconds(FADE_DURATION);
        fadePanel.gameObject.SetActive(false);
        
        isInteriorView = true;
        isTransitioning = false;

        if (aiDirector.HasPausedEvents()) {
            aiDirector.ResumeEvents();
        }

```csharp
/// <summary>
/// Calculates radiation protection factor based on material selection and thickness
/// </summary>
/// <param name="materials">Dictionary of materials and thicknesses</param>
/// <returns>Radiation protection percentage (0-100)</returns>
public float CalculateRadiationProtection(Dictionary<HabitatMaterial, float> materials)
{
    float totalProtection = 0f;
    
    foreach (KeyValuePair<HabitatMaterial, float> material in materials)
    {
        // Get base radiation stopping power per cm for this material
        float baseProtection = GetMaterialRadiationStoppingPower(material.Key);
        
        // Apply thickness modifier (diminishing returns)
        float effectiveThickness = Mathf.Min(material.Value, MAX_EFFECTIVE_THICKNESS);
        float thicknessModifier = Mathf.Log10(1f + effectiveThickness) / Mathf.Log10(1f + MAX_EFFECTIVE_THICKNESS);
        
        // Calculate protection contribution
        float materialProtection = baseProtection * thicknessModifier;
        
        // Add to total using diminishing returns formula for layered protection
        totalProtection = totalProtection + materialProtection * (1f - totalProtection / 100f);
    }
    
    return Mathf.Clamp(totalProtection, 0f, MAX_RADIATION_PROTECTION);
}

/// <summary>
/// Calculates construction time based on habitat type, size, and construction method
/// </summary>
/// <param name="type">Type of habitat</param>
/// <param name="volume">Total habitable volume in cubic meters</param>
/// <param name="constructionMethod">Method of construction</param>
/// <returns>Construction time in days</returns>
public float CalculateConstructionTime(
    HabitatType type, 
    float volume, 
    ConstructionMethod constructionMethod)
{
    // Base construction rates (cubic meters per day)
    float baseRate = 0f;
    switch (constructionMethod)
    {
        case ConstructionMethod.Prefabricated:
            baseRate = PREFAB_CONSTRUCTION_RATE;
            break;
        case ConstructionMethod.Inflatable:
            baseRate = INFLATABLE_CONSTRUCTION_RATE;
            break;
        case ConstructionMethod.InSituPrinting:
            baseRate = PRINTING_CONSTRUCTION_RATE;
            break;
        case ConstructionMethod.RoboticAssembly:
            baseRate = ROBOTIC_CONSTRUCTION_RATE;
            break;
    }
    
    // Apply type-specific modifiers
    float typeModifier = 1f;
    switch (type)
    {
        case HabitatType.RigidModule:
            typeModifier = RIGID_CONSTRUCTION_MODIFIER;
            break;
        case HabitatType.InflatableHabitat:
            typeModifier = INFLATABLE_CONSTRUCTION_MODIFIER;
            break;
        case HabitatType.RotatingStructure:
            typeModifier = ROTATING_CONSTRUCTION_MODIFIER;
            break;
    }
    
    // Calculate construction time with increasing difficulty for larger volumes
    float scaledVolume = volume * Mathf.Log10(1f + volume / 1000f);
    float constructionTime = scaledVolume / (baseRate * typeModifier);
    
    return Mathf.Max(constructionTime, MINIMUM_CONSTRUCTION_TIME);
}

/// <summary>
/// Calculates required habitable space based on population and mission parameters
/// </summary>
/// <param name="populationCount">Number of inhabitants</param>
/// <param name="missionDuration">Mission duration in years</param>
/// <param name="hasArtificialGravity">Whether artificial gravity is available</param>
/// <returns>Space requirements in square meters</returns>
public HabitableSpaceRequirements CalculateHabitableSpace(
    int populationCount,
    float missionDuration,
    bool hasArtificialGravity)
{
    HabitableSpaceRequirements requirements = new HabitableSpaceRequirements();
    
    // Base personal space (private quarters)
    float basePersonalSpace = BASE_PERSONAL_SPACE;
    
    // Long duration missions require more personal space
    if (missionDuration > 1f)
    {
        basePersonalSpace *= 1f + (missionDuration - 1f) * DURATION_SPACE_FACTOR;
    }
    
    // Artificial gravity allows more efficient space usage
    if (hasArtificialGravity)
    {
        basePersonalSpace *= GRAVITY_SPACE_EFFICIENCY;
    }
    
    // Calculate private space total
    requirements.privateSpace = populationCount * basePersonalSpace;
    
    // Community space scales sub-linearly with population
    requirements.communitySpace = COMMUNITY_BASE_SPACE + 
        (populationCount * COMMUNITY_SPACE_PER_PERSON * 
        Mathf.Pow(populationCount, COMMUNITY_SCALING_EXPONENT));
    
    // Technical space for equipment (20% of habitable)
    float habitableSpace = requirements.privateSpace + requirements.communitySpace;
    requirements.technicalSpace = habitableSpace * TECHNICAL_SPACE_RATIO;
    
    // Total space
    requirements.totalSpace = habitableSpace + requirements.technicalSpace;
    
    // Calculate morale impact of space adequacy
    float spaceAdequacy = (requirements.privateSpace / populationCount) / IDEAL_PERSONAL_SPACE;
    requirements.moraleImpact = Mathf.Clamp(
        (spaceAdequacy - 0.5f) * 2f * MAX_SPACE_MORALE_IMPACT,
        -MAX_SPACE_MORALE_IMPACT,
        MAX_SPACE_MORALE_IMPACT
    );
    
    return requirements;
}

/// <summary>
/// Calculates artificial gravity parameters for rotating habitats
/// </summary>
/// <param name="desiredGravityLevel">Desired gravity as fraction of Earth gravity (0-1)</param>
/// <param name="maxRotationRate">Maximum comfortable rotation rate in RPM</param>
/// <returns>Rotation parameters for the habitat</returns>
public RotationParameters CalculateRotationParameters(
    float desiredGravityLevel,
    float maxRotationRate)
{
    RotationParameters parameters = new RotationParameters();
    
    // Convert desired gravity to m/s²
    float targetAcceleration = desiredGravityLevel * EARTH_GRAVITY;
    
    // Convert max rotation rate to rad/s
    float maxAngularVelocity = maxRotationRate * RPM_TO_RADIANS;
    
    // Calculate minimum radius to achieve desired gravity at max rotation rate
    // a = ω²r
    float minimumRadius = targetAcceleration / (maxAngularVelocity * maxAngularVelocity);
    
    // If minimum radius is below comfort threshold, increase radius and reduce rotation rate
    if (minimumRadius < MINIMUM_COMFORT_RADIUS)
    {
        minimumRadius = MINIMUM_COMFORT_RADIUS;
        // ω = sqrt(a/r)
        maxAngularVelocity = Mathf.Sqrt(targetAcceleration / minimumRadius);
    }
    
    // Set output parameters
    parameters.radius = minimumRadius;
    parameters.rotationRate = maxAngularVelocity * RADIANS_TO_RPM;
    parameters.targetGravity = desiredGravityLevel;
    
    // Calculate Coriolis effect strength (0-1 scale)
    // Stronger at higher rotation rates and lower radii
    parameters.coriolisStrength = Mathf.Clamp01(
        (parameters.rotationRate / MAX_COMFORTABLE_RPM) * 
        (REFERENCE_CORIOLIS_RADIUS / parameters.radius)
    );
    
    return parameters;
}

```csharp
/// <summary>
/// Simulates micrometeorite impact on habitat structure
/// </summary>
/// <param name="habitat">The affected habitat</param>
/// <param name="impactEnergy">Impact energy in joules</param>
/// <param name="targetSection">Section of the habitat that was hit</param>
/// <returns>Impact result including damage assessment</returns>
public ImpactResult SimulateMicrometeoroidImpact(
    Habitat habitat,
    float impactEnergy,
    HabitatSection targetSection)
{
    ImpactResult result = new ImpactResult();
    
    // Get shield effectiveness for the target section
    float shieldEffectiveness = habitat.GetShieldEffectiveness(targetSection);
    
    // Calculate penetration threshold
    float penetrationThreshold = targetSection.GetPenetrationThreshold();
    
    // Reduce impact energy based on shielding
    float attenuatedEnergy = impactEnergy * (1f - shieldEffectiveness);
    
    // Determine if penetration occurred
    bool penetration = attenuatedEnergy > penetrationThreshold;
    result.penetration = penetration;
    
    if (penetration)
    {
        // Calculate breach size
        result.breachSize = CalculateBreachSize(attenuatedEnergy, penetrationThreshold);
        
        // Initialize pressure loss simulation
        result.pressureLossRate = CalculatePressureLossRate(
            result.breachSize,
            targetSection.GetVolume(),
            habitat.GetCurrentPressure()
        );
        
        // Determine if automatic sealing is possible
        result.canAutoSeal = result.breachSize < targetSection.GetAutoSealThreshold();
        
        // Calculate repair requirements
        result.repairTimeRequired = CalculateRepairTime(result.breachSize);
        result.repairMaterialsRequired = CalculateRepairMaterials(result.breachSize);
        
        // Set up cascading failures
        result.cascadingFailures = DetermineCascadingFailures(
            targetSection,
            attenuatedEnergy,
            habitat.GetSystemsInSection(targetSection)
        );
    }
    else
    {
        // Non-penetrating impact still causes shield damage
        result.shieldDamage = CalculateShieldDamage(attenuatedEnergy, penetrationThreshold);
    }
    
    return result;
}

/// <summary>
/// Calculates system reliability based on redundancy level and component MTBF
/// </summary>
/// <param name="systemType">Type of system</param>
/// <param name="redundancyLevel">Level of redundancy (1=no redundancy, 2=dual, 3=triple)</param>
/// <param name="componentMTBF">Mean time between failures for individual component in hours</param>
/// <param name="hoursOfOperation">Total hours the system has been operating</param>
/// <returns>Current reliability percentage (0-100)</returns>
public float CalculateSystemReliability(
    SystemType systemType,
    int redundancyLevel,
    float componentMTBF,
    float hoursOfOperation)
{
    // Base reliability using exponential failure model
    float baseReliability = Mathf.Exp(-hoursOfOperation / componentMTBF);
    
    // System-specific modifier
    float systemModifier = GetSystemReliabilityModifier(systemType);
    baseReliability *= systemModifier;
    
    // Apply redundancy effect
    float systemReliability = 0f;
    
    if (redundancyLevel <= 1)
    {
        // No redundancy
        systemReliability = baseReliability;
    }
    else
    {
        // Calculate probability of all redundant units failing
        // R = 1 - (1-r)^n where r is component reliability and n is redundancy level
        systemReliability = 1f - Mathf.Pow(1f - baseReliability, redundancyLevel);
    }
    
    return systemReliability * 100f; // Convert to percentage
}
```

4.11 Time Progression System
---------------------------

The Time Progression System implements a hybrid timeline approach that combines continuous player-controlled acceleration with narrative-driven time skips. This system forms the core of the game's temporal mechanics, enabling both moment-to-moment gameplay and generational progression.

### 4.11.1 System Architecture

The Time Progression System is built around the following core components:

```csharp
/// <summary>
/// Manages the game's time progression, implementing a hybrid timeline approach
/// that combines continuous player-controlled time flow with narrative-driven time skips.
/// </summary>
public class TimeController : MonoBehaviour
{
    // Singleton instance
    public static TimeController Instance { get; private set; }
    
    // Time scale configuration
    [SerializeField] private float _baseTimeScale = 1.0f;
    [SerializeField] private float[] _timeAccelerationLevels = { 1.0f, 5.0f, 20.0f, 100.0f, 500.0f };
    [SerializeField] private float _minTimeScale = 0.0f;
    [SerializeField] private float _maxTimeScale = 500.0f;
    
    // Current state
    private int _currentTimeScaleIndex = 0;
    private float _currentTimeScale = 1.0f;
    private int _savedTimeScaleIndex = 0;
    private bool _isPaused = false;
    private bool _timeSkipInProgress = false;
    
    // Game time tracking
    private DateTime _currentGameDate;
    [SerializeField] private DateTime _startGameDate = new DateTime(2135, 1, 1);
    private float _secondsPerGameDay = 60.0f; // At 1x speed, one game day passes in 60 real seconds
    private float _accumulatedRealTime = 0.0f;
    
    // Time skip state
    private TimeSkipRequest _pendingSkipRequest = null;
    private List<TimelineMilestone> _timeSkipMilestones = new List<TimelineMilestone>();
    private int _currentMilestoneIndex = 0;
    
    // Events
    public event Action<float> OnTimeScaleChanged;
    public event Action<DateTime> OnDateChanged;
    public event Action<TimeSkipRequest> OnTimeSkipRequested;
    public event Action<TimelineMilestone> OnTimeSkipMilestoneReached;
    public event Action<DateTime> OnTimeSkipCompleted;
    
    private void Awake()
    {
        // Singleton pattern
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }
        Instance = this;
        
        // Initialize date
        _currentGameDate = _startGameDate;
        
        // Listen for events
        GameEvents.TimeSkipMilestoneResolved.AddListener(OnMilestoneResolved);
    }
    
    private void Update()
    {
        if (_timeSkipInProgress)
        {
            // Skip is active but waiting for milestone resolution
            if (_currentMilestoneIndex < _timeSkipMilestones.Count && 
                _timeSkipMilestones[_currentMilestoneIndex].RequiresPlayerIntervention)
            {
                return; // Waiting for player to resolve current milestone
            }
            
            ProcessTimeSkip();
        }
        else if (!_isPaused)
        {
            // Normal time progression
            UpdateGameTime(Time.deltaTime);
        }
    }
    
    /// <summary>
    /// Updates the game time based on elapsed real time and current time scale
    /// </summary>
    private void UpdateGameTime(float deltaTime)
    {
        // Calculate time advancement
        float scaledDeltaTime = deltaTime * _currentTimeScale;
        _accumulatedRealTime += scaledDeltaTime;
        
        // Calculate how many game days have passed
        float gameTimeDaysPassed = _accumulatedRealTime / _secondsPerGameDay;
        
        if (gameTimeDaysPassed >= 0.01f) // Only update if at least 0.01 days have passed
        {
            // Advance game time
            _currentGameDate = _currentGameDate.AddDays(gameTimeDaysPassed);
            _accumulatedRealTime -= gameTimeDaysPassed * _secondsPerGameDay;
            
            // Notify listeners
            OnDateChanged?.Invoke(_currentGameDate);
            
            // Check for events at this time
            TimelineEventManager.Instance.CheckForScheduledEvents(_currentGameDate);
        }
    }
    
    /// <summary>
    /// Sets the time scale to a specific index in the acceleration levels array
    /// </summary>
    public void SetTimeScaleIndex(int index)
    {
        if (index >= 0 && index < _timeAccelerationLevels.Length)
        {
            _currentTimeScaleIndex = index;
            _currentTimeScale = _timeAccelerationLevels[index] * _baseTimeScale;
            OnTimeScaleChanged?.Invoke(_currentTimeScale);
        }
    }
    
    /// <summary>
    /// Increases the time scale to the next acceleration level
    /// </summary>
    public void IncreaseTimeScale()
    {
        if (_currentTimeScaleIndex < _timeAccelerationLevels.Length - 1)
        {
            SetTimeScaleIndex(_currentTimeScaleIndex + 1);
        }
    }
    
    /// <summary>
    /// Decreases the time scale to the previous acceleration level
    /// </summary>
    public void DecreaseTimeScale()
    {
        if (_currentTimeScaleIndex > 0)
        {
            SetTimeScaleIndex(_currentTimeScaleIndex - 1);
        }
    }
    
    /// <summary>
    /// Pauses the game time progression
    /// </summary>
    public void PauseTime()
    {
        if (!_isPaused)
        {
            _isPaused = true;
            _savedTimeScaleIndex = _currentTimeScaleIndex;
            SetTimeScaleIndex(0); // Set to 0 index (which is 1.0f, but gets multiplied by 0 effectively)
        }
    }
    
    /// <summary>
    /// Resumes the game time progression
    /// </summary>
    public void ResumeTime()
    {
        if (_isPaused)
        {
            _isPaused = false;
            SetTimeScaleIndex(_savedTimeScaleIndex);
        }
    }
    
    /// <summary>
    /// Requests a significant time skip
    /// </summary>
    public void RequestTimeSkip(TimeSpan skipDuration, TimeSkipConditions conditions)
    {
        if (_timeSkipInProgress)
            return;
        
        // Create time skip request
        _pendingSkipRequest = new TimeSkipRequest
        {
            Duration = skipDuration,
            TargetDate = _currentGameDate.Add(skipDuration),
            Conditions = conditions,
            InterruptConditionsMet = false
        };
        
        // Generate milestone events for this time skip
        _timeSkipMilestones = TimelineMilestoneGenerator.GenerateForTimespan(
            _currentGameDate, 
            _pendingSkipRequest.TargetDate,
            GameStateManager.Instance.CurrentGameState
        );
        
        // Notify UI of time skip request
        OnTimeSkipRequested?.Invoke(_pendingSkipRequest);
    }
    
    /// <summary>
    /// Begins the time skip execution
    /// </summary>
    public void ExecuteTimeSkip()
    {
        if (_pendingSkipRequest == null)
            return;
        
        // Save current time scale and pause normal time
        _savedTimeScaleIndex = _currentTimeScaleIndex;
        PauseTime();
        
        // Start skip
        _timeSkipInProgress = true;
        _currentMilestoneIndex = 0;
        
        // Begin processing skip
        ProcessTimeSkip();
    }
    
    /// <summary>
    /// Processes the current time skip
    /// </summary>
    private void ProcessTimeSkip()
    {
        // Check if there are milestones to process
        while (_currentMilestoneIndex < _timeSkipMilestones.Count)
        {
            TimelineMilestone milestone = _timeSkipMilestones[_currentMilestoneIndex];
            
            // If we haven't reached this milestone's time yet, advance time to it
            if (milestone.OccurrenceDate > _currentGameDate)
            {
                // Jump to this milestone's time
                _currentGameDate = milestone.OccurrenceDate;
                OnDateChanged?.Invoke(_currentGameDate);
            }
            
            // Process the milestone
            milestone.ApplyImmediateChanges();
            
            // Record for history
            TimelineHistoryManager.Instance.RecordMilestone(milestone);
            
            // If this milestone requires player intervention, pause here
            if (milestone.RequiresPlayerIntervention)
            {
                OnTimeSkipMilestoneReached?.Invoke(milestone);
                return; // Wait for player to resolve this milestone
            }
            
            // Move to next milestone
            _currentMilestoneIndex++;
        }
        
        // If we've processed all milestones, complete the skip
        CompleteTimeSkip();
    }
    
    /// <summary>
    /// Called when a milestone requiring player intervention has been resolved
    /// </summary>
    private void OnMilestoneResolved()
    {
        // Move to next milestone
        _currentMilestoneIndex++;
        
        // Continue processing
        ProcessTimeSkip();
    }
    
    /// <summary>
    /// Completes the current time skip
    /// </summary>
    private void CompleteTimeSkip()
    {
        // Jump to target date
        _currentGameDate = _pendingSkipRequest.TargetDate;
        
        // Apply final changes
        GameStateManager.Instance.ApplyTimeSkipEffects(_pendingSkipRequest.Duration);
        
        // Cleanup
        _timeSkipInProgress = false;
        _pendingSkipRequest = null;
        _timeSkipMilestones.Clear();
        
        // Notify listeners
        OnDateChanged?.Invoke(_currentGameDate);
        OnTimeSkipCompleted?.Invoke(_currentGameDate);
        
        // Resume previous time scale
        ResumeLastTimeScale();
    }
    
    /// <summary>
    /// Cancels the current time skip
    /// </summary>
    public void CancelTimeSkip()
    {
        if (!_timeSkipInProgress)
            return;
        
        // Cleanup
        _timeSkipInProgress = false;
        _pendingSkipRequest = null;
        _timeSkipMilestones.Clear();
        
        // Resume previous time scale
        ResumeLastTimeScale();
    }
    
    /// <summary>
    /// Resumes the last active time scale before pausing
    /// </summary>
    public void ResumeLastTimeScale()
    {
        _isPaused = false;
        SetTimeScaleIndex(_savedTimeScaleIndex);
    }
    
    /// <summary>
    /// Returns the current game date
    /// </summary>
    public DateTime GetCurrentDate()
    {
        return _currentGameDate;
    }
    
    /// <summary>
    /// Returns the current time scale
    /// </summary>
    public float GetCurrentTimeScale()
    {
        return _currentTimeScale;
    }
}
```

    /// Applies this choice's effects to the game state
    /// </summary>

        ResourceManager resourceManager = ResourceManager.Instance;
        foreach (var impact in ResourceImpacts)

            resourceManager.ModifyResource(impact.Key, impact.Value);
        }
        
        // Apply crew impacts
        CrewManager crewManager = CrewManager.Instance;
        foreach (var impact in CrewImpacts)
        {
            string[] parts = impact.Key.Split('.');
            if (parts.Length == 2)
            {
                string crewID = parts[0];
                string attribute = parts[1];
                crewManager.ModifyCrewAttribute(crewID, attribute, impact.Value);
            }
        }
    }

### 4.11.4 Timeline History Management

The system maintains a historical record of significant events:

```csharp
/// <summary>
/// Manages the history of timeline events for record-keeping and UI display
/// </summary>
public class TimelineHistoryManager : MonoBehaviour
{
    // Singleton instance
    public static TimelineHistoryManager Instance { get; private set; }
    
    // Historical data
    private List<TimelineHistoricalEvent> _historicalEvents = new List<TimelineHistoricalEvent>();
    private Dictionary<string, List<TimelineHistoricalEvent>> _eventsByCategory = new Dictionary<string, List<TimelineHistoricalEvent>>();
    
    // Maximum history size
    [SerializeField] private int _maxHistorySize = 1000;
    
    private void Awake()
    {
        // Singleton pattern
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }
        Instance = this;
    }
    
    /// <summary>
    /// Records a milestone event in the timeline history
    /// </summary>
    public void RecordMilestone(TimelineMilestone milestone)
    {
        TimelineHistoricalEvent histEvent = new TimelineHistoricalEvent
        {
            ID = milestone.ID,
            Title = milestone.Title,
            Description = milestone.Description,
            OccurrenceDate = milestone.OccurrenceDate,
            Type = milestone.Type.ToString(),
            Category = GetCategoryForMilestoneType(milestone.Type)
        };
        
        AddEventToHistory(histEvent);
    }
    
    /// <summary>
    /// Records a player choice in the timeline history
    /// </summary>
    public void RecordChoice(TimelineMilestone milestone, MilestoneChoice choice)
    {
        TimelineHistoricalEvent histEvent = new TimelineHistoricalEvent
        {
            ID = choice.ID,
            Title = "Decision: " + milestone.Title,
            Description = "Choice made: " + choice.Text + "\n\n" + choice.Description,
            OccurrenceDate = milestone.OccurrenceDate,
            Type = "PlayerChoice",
            Category = "Decisions",
            RelatedEventID = milestone.ID
        };
        
        AddEventToHistory(histEvent);
    }
    
    /// <summary>
    /// Adds an event to the history, maintaining size limits
    /// </summary>
    private void AddEventToHistory(TimelineHistoricalEvent histEvent)
    {
        // Add to main list
        _historicalEvents.Add(histEvent);
        
        // Add to category dictionary
        if (!_eventsByCategory.ContainsKey(histEvent.Category))
        {
            _eventsByCategory[histEvent.Category] = new List<TimelineHistoricalEvent>();
        }
        _eventsByCategory[histEvent.Category].Add(histEvent);
        
        // Sort by date
        _historicalEvents.Sort((a, b) => a.OccurrenceDate.CompareTo(b.OccurrenceDate));
        
        // Trim if over size limit
        if (_historicalEvents.Count > _maxHistorySize)
        {
            TimelineHistoricalEvent oldestEvent = _historicalEvents[0];
            _historicalEvents.RemoveAt(0);
            
            // Also remove from category list
            if (_eventsByCategory.ContainsKey(oldestEvent.Category))
            {
                _eventsByCategory[oldestEvent.Category].Remove(oldestEvent);
            }
        }
    }
    
    /// <summary>
    /// Returns all historical events
    /// </summary>
    public List<TimelineHistoricalEvent> GetHistoricalEvents()
    {
        return new List<TimelineHistoricalEvent>(_historicalEvents);
    }
    
    /// <summary>
    /// Returns historical events filtered by category
    /// </summary>
    public List<TimelineHistoricalEvent> GetEventsByCategory(string category)
    {
        if (_eventsByCategory.ContainsKey(category))
        {
            return new List<TimelineHistoricalEvent>(_eventsByCategory[category]);
        }
        return new List<TimelineHistoricalEvent>();
    }
    
    /// <summary>
    /// Maps milestone types to categories for organization
    /// </summary>
    private string GetCategoryForMilestoneType(MilestoneType type)
    {
        switch (type)
        {
            case MilestoneType.CrewEvent:
                return "Crew";
            case MilestoneType.ShipEvent:
                return "Ship";
            case MilestoneType.ResourceEvent:
                return "Resources";
            case MilestoneType.FactionEvent:
                return "Factions";
            case MilestoneType.ResearchEvent:
                return "Research";
            case MilestoneType.StoryEvent:
                return "Story";
            case MilestoneType.RandomEvent:
                return "Misc";
            default:
                return "General";
        }
    }
}

/// <summary>
/// Represents a historical timeline event for record-keeping
/// </summary>
public class TimelineHistoricalEvent
{
    public string ID { get; set; }
    public string Title { get; set; }
    public string Description { get; set; }
    public DateTime OccurrenceDate { get; set; }
    public string Type { get; set; }
    public string Category { get; set; }
    public string RelatedEventID { get; set; }
}
