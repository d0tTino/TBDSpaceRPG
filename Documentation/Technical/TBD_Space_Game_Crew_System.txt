1.0 Crew System Overview
========================
This document details the comprehensive Crew System for TBD Space Game. The Crew System is the backbone of the gameplay experience, simulating the lives, relationships, and evolution of space colonists across multiple generations.

Core features of the Crew System:
- Dynamic character generation with genetic inheritance
- Complex skill and attribute development
- Cultural identity evolution across generations
- Relationship networks and social dynamics
- Crew management mechanics and interfaces
- Integration with other game systems

2.0 Character Generation
========================
The Crew System incorporates a sophisticated character generation system that creates diverse crew members with unique traits, skills, and backgrounds.

2.1 Crew Design
-------------
Each crew member is defined by three core components: roles, traits, and attributes. These elements determine how crew members perform tasks, interact with each other, and evolve over time.

### Roles
Crew members are assigned specific roles that tie directly to ship systems and tasks:
- Pilot: Responsible for navigation and maneuvering the spacecraft
- Engineer: Manages repairs, maintenance, and system upgrades
- Scientist: Conducts research and handles scientific equipment
- Medic: Maintains crew health and manages medical emergencies
- Communications Officer: Handles external communications and diplomacy

### Traits
Each crew member has one or more personality traits that influence their behavior and task performance. Traits are static and do not change:
- Hot-headed: Increases the likelihood of conflicts but boosts performance under pressure
- Genius: Enhances problem-solving, increasing the success rate of complex tasks
- Calm: Reduces stress during crises, improving decision-making
- Empathetic: Boosts morale of nearby crew members but may reduce efficiency in high-stress situations

### Background Influence
A crew member's background influences their traits and initial attributes:

```csharp
// Sample code for generating crew traits based on background
public void AssignTraits(CrewMember crew)
{
    List<Trait> availableTraits = traitDatabase.GetAllTraits();
    List<Trait> assignedTraits = new List<Trait>();
    
    // Base number of traits
    int traitCount = Random.Range(MIN_TRAITS_PER_CREW, MAX_TRAITS_PER_CREW + 1);
    
    // Apply background modifiers to trait probabilities
    Dictionary<string, float> traitProbabilities = new Dictionary<string, float>();
    foreach (Trait trait in availableTraits)
    {
        // Start with base probability
        traitProbabilities[trait.id] = trait.baseProbability;
    }
    
    // Modify based on background
    if (crew.Background == CrewBackground.EarthBorn)
    {
        // Faction-specific modifications
        if (crew.Faction == Faction.US)
        {
            traitProbabilities["Individualist"] *= 1.15f;
            traitProbabilities["Innovative"] *= 1.10f;
        }
        else if (crew.Faction == Faction.China)
        {
            traitProbabilities["Collectivist"] *= 1.15f;
            traitProbabilities["Disciplined"] *= 1.10f;
        }
        else
        {
            traitProbabilities["Adaptable"] *= 1.15f;
            traitProbabilities["SelfReliant"] *= 1.10f;
        }
    }
    else if (crew.GenerationNumber == 1)
    {
        // First generation space-born
        traitProbabilities["LowGAdapted"] *= 1.25f;
        traitProbabilities["ConfinedSpaceComfort"] *= 1.15f;
        traitProbabilities["EarthNostalgia"] *= 0.85f;
        traitProbabilities["ResourceConscious"] *= 1.10f;
    }
    else
    {
        // Second+ generation space-born
        traitProbabilities["MissionSkeptic"] *= 1.30f;
        traitProbabilities["HeritagePreserver"] *= 1.20f;
        
        // Chance for new emergent traits
        if (Random.value < 0.25f)
        {
            Trait newTrait = GenerateEmergentTrait(crew.Environment);
            assignedTraits.Add(newTrait);
            traitCount--;
        }
    }
    
    // Assign traits based on modified probabilities
    for (int i = 0; i < traitCount; i++)
    {
        Trait selectedTrait = SelectTraitBasedOnProbability(availableTraits, traitProbabilities);
        assignedTraits.Add(selectedTrait);
        
        // Reduce probability to avoid duplicates of similar traits
        foreach (string conflictingTraitId in selectedTrait.conflictingTraits)
        {
            traitProbabilities[conflictingTraitId] *= 0.5f;
        }
    }
    
    // Apply traits to crew member
    foreach (Trait trait in assignedTraits)
    {
        crew.AddTrait(trait);
    }
    
    // Set crisis response patterns based on background
    if (crew.Background == CrewBackground.EarthBorn)
    {
        crew.efficiencyModifier = 0.15f;
        crew.stressResistanceModifier = -0.05f;
    }
    else if (crew.GenerationNumber == 1)
    {
        crew.efficiencyModifier = 0.05f;
        crew.stressResistanceModifier = 0.10f;
    }
    else
    {
        crew.efficiencyModifier = -0.05f;
        crew.stressResistanceModifier = 0.20f;
    }
}
```

```csharp
void UpdateExperienceAndLearning(CrewMember crew, float deltaTime) {
    // Check if crew is performing a skill-based action
    if (crew.currentAction != null && crew.currentAction.providesExperience) {
        // Accumulate experience in the relevant skill
        SkillType relevantSkill = crew.currentAction.relatedSkill;
        crew.AddExperience(relevantSkill, EXPERIENCE_GAIN_RATE * deltaTime);
        
        // Chance to improve at task when meaningful experience threshold is reached
        if (crew.GetExperience(relevantSkill) >= SKILL_IMPROVEMENT_THRESHOLD) {
            // Improve skill and reset experience counter
            crew.ImproveSkill(relevantSkill, SKILL_IMPROVEMENT_AMOUNT);
            crew.ResetExperience(relevantSkill);
            
            // Notify player of skill improvement
            notificationSystem.CreateSkillImprovedNotification(crew, relevantSkill);
        }
    }
    
    // Learning from observation (watching others perform tasks)
    if (crew.GetState() == CrewState.Observing && crew.observingAction != null) {
        CrewMember observedCrew = crew.observingCrewMember;
        SkillType observedSkill = crew.observingAction.relatedSkill;
        
        // Gain experience at a reduced rate when observing
        float observationExperienceRate = OBSERVATION_EXPERIENCE_FACTOR * 
                                       EXPERIENCE_GAIN_RATE * 
                                       deltaTime;
        
        crew.AddExperience(observedSkill, observationExperienceRate);
    }
}
```

```csharp
{

    foreach (CrewMember crew in crewManager.GetAllCrewMembers())
    {

        float gravityLevel = environmentManager.GetCurrentGravity();
        float radiationLevel = environmentManager.GetCurrentRadiationLevel();

        float boneLossRate = BASE_BONE_LOSS_RATE * (1f - gravityLevel);
        float muscleAtrophyRate = BASE_MUSCLE_ATROPHY_RATE * (1f - gravityLevel);
        float cardioDeclineRate = BASE_CARDIO_DECLINE_RATE * (1f - gravityLevel);

        if (crew.HasActiveTreatment(TreatmentType.BoneDensityTherapy))
        {
            boneLossRate *= (1f - crew.GetTreatmentEfficacy(TreatmentType.BoneDensityTherapy));
        }
        
        if (crew.HasActiveTreatment(TreatmentType.MuscleStimulation))
        {
            muscleAtrophyRate *= (1f - crew.GetTreatmentEfficacy(TreatmentType.MuscleStimulation));
        }
        
        // Apply exercise regimen benefits if compliance is high
        if (crew.exerciseCompliance > EXERCISE_COMPLIANCE_THRESHOLD)

            muscleAtrophyRate *= (1f - EXERCISE_MUSCLE_PROTECTION);
            cardioDeclineRate *= (1f - EXERCISE_CARDIO_PROTECTION);

        }
```

4.2 Cultural Evolution
-------------------
Cultural identity evolves over time, particularly in generation ships and permanent settlements:

### Cultural Divergence Mechanics
Cultural evolution follows a timeline based on years since departure from Earth:
- Language: New terminology (5-10 years), distinct slang (15-25 years), full dialect (50-100 years)
- Customs: Adapted routines (3-5 years), new celebrations (10-15 years), distinct rituals (30-40 years)
- Values: Priority shifts (10-20 years), philosophical divergence (40-60 years), new ethical frameworks (80-100 years)

```csharp
CulturalDivergence CalculateCulturalDivergence(Colony colony) {
    int yearsSinceFoundation = worldState.GetCurrentYear() - colony.foundingYear;
    
    // Calculate base divergence in various categories
    float languageDivergence = CalculateLogisticDivergence(yearsSinceFoundation, 75f, 0.07f);
    float customDivergence = CalculateLogisticDivergence(yearsSinceFoundation, 40f, 0.09f);
    float valueDivergence = CalculateLogisticDivergence(yearsSinceFoundation, 90f, 0.05f);
    
    // Adjust based on Earth communication frequency
    float earthCommsFrequency = colony.GetEarthCommunicationFrequency();
    float communicationModifier = Mathf.Lerp(1.5f, 0.5f, Mathf.Clamp01(earthCommsFrequency));
    
    languageDivergence *= communicationModifier;
    customDivergence *= communicationModifier;
    valueDivergence *= communicationModifier;
    
    // Adjust based on cultural preservation efforts
    float preservationFactor = colony.GetCulturalPreservationFactor();
    float preservationModifier = Mathf.Lerp(1.2f, 0.6f, Mathf.Clamp01(preservationFactor));
    
    languageDivergence *= preservationModifier;
    customDivergence *= preservationModifier;
    valueDivergence *= preservationModifier;
    
    // Return combined divergence metrics
    return new CulturalDivergence {
        language = languageDivergence,
        customs = customDivergence,
        values = valueDivergence
    };
}
```

### Generational Evolution Mechanics
Cultural identity evolves across generations through a sophisticated inheritance and adaptation system:

```csharp
/// <summary>
/// Evolves cultural identity across generations based on environmental factors.
/// </summary>
/// <param name="parentIdentities">The cultural identities of the parents.</param>
/// <param name="generationNumber">Which generation from Earth (1-based).</param>
/// <param name="environmentFactors">Environmental influences shaping development.</param>
/// <returns>A new cultural identity for the offspring.</returns>
public CulturalIdentity EvolveIdentityForOffspring(
    List<CulturalIdentity> parentIdentities,
    int generationNumber,
    EnvironmentFactors environmentFactors)
{
    CulturalIdentity newIdentity = new CulturalIdentity();
    
    // Base inheritance from parents
    foreach (ValueAxis axis in ValueAxis.All)
    {
        float averageValue = parentIdentities.Average(p => p.ValueSystem[axis]);
        
        // Apply generational drift
        float generationalDrift = CalculateGenerationalDrift(
            axis, 
            generationNumber, 
            environmentFactors
        );
        
        newIdentity.ValueSystem[axis] = Mathf.Clamp(
            averageValue + generationalDrift,
            -1f,
            1f
        );
    }
    
    // Belief system evolution
    if (generationNumber == 1)
    {
        // First generation mostly inherits directly
        newIdentity.PrimaryBelief = SelectDominantBeliefSystem(parentIdentities);
        newIdentity.BeliefCommitment = CalculateFirstGenBeliefCommitment(parentIdentities);
    }
    else if (generationNumber == 2)
    {
        // Second generation shows more pragmatic adaptation
        newIdentity.PrimaryBelief = ApplyEnvironmentalInfluence(
            parentIdentities,
            environmentFactors
        );
        newIdentity.BeliefCommitment = CalculateSecondGenBeliefCommitment(
            parentIdentities,
            environmentFactors
        );
    }
    else
    {
        // Third+ generation develops genuinely new belief systems
        newIdentity.PrimaryBelief = GenerateSpaceBornBelief(
            parentIdentities,
            generationNumber,
            environmentFactors
        );
        newIdentity.BeliefCommitment = CalculateLateCulturalBeliefCommitment(
            generationNumber,
            environmentFactors
        );
    }
    
    // Generate appropriate rituals
    newIdentity.Rituals = GenerateEvolvingRituals(
        parentIdentities,
        newIdentity.PrimaryBelief,
        generationNumber,
        environmentFactors
    );
    
    return newIdentity;
}
```

```csharp
// Cultural event system
public class CulturalEvent
{
    public string Name { get; private set; }
    public string Description { get; private set; }
    public float ResourceCost { get; private set; }
    public float SpaceRequirement { get; private set; }
    public float BaseParticipationRate { get; private set; }
    public float CohesionImpact { get; private set; }
    public List<string> RelatedCultures { get; private set; }
    public bool IsReligious { get; private set; }
    public string RelatedReligion { get; private set; }
    
    // Calculate actual participation based on crew composition
    public float CalculateExpectedParticipation(List<CrewMember> crew)
    {
        float totalParticipation = 0f;
        
        foreach (var member in crew)
        {
            float individualLikelihood = BaseParticipationRate;
            
            // Cultural alignment boost
            if (RelatedCultures.Contains(member.CulturalBackground))
            {
                individualLikelihood += 0.3f;
            }
            
            // Religious alignment
            if (IsReligious && member.ReligiousBackground == RelatedReligion)
            {
                individualLikelihood += 0.4f * member.ReligiousDevoutness;
            }
            
            // Space-born adjustment
            if (member.IsBornInSpace && !IsReligious)
            {
                // Space-born crew prefer space-developed events
                individualLikelihood += Name.Contains("Space") ? 0.2f : -0.1f;
            }
            
            totalParticipation += Mathf.Clamp01(individualLikelihood);
        }
        
        return totalParticipation / crew.Count;
    }
}
```

4.5 Cultural Conflict and Resolution
---------------------------------
The system models cultural conflicts and resolutions, creating gameplay moments where players must navigate complex cultural dynamics:

### Conflict Triggers
- Value clashes on key decisions (e.g., resource allocation, diplomatic approaches)
- Religious observance disputes (e.g., use of shared spaces, time allocation)
- Generational divides between Earth-born and space-born crew members

### Resolution Mechanics
- Dialogue-based mediation with options shaped by crew member traits
- Policy establishment (e.g., religious observance guidelines, cultural sensitivity training)
- Cultural exchange events that build understanding

### Impact on Gameplay
- Ship morale affected by unresolved cultural conflicts (-10% efficiency per major conflict)
- Decision-making influenced by dominant cultural frameworks
- Diplomatic options shaped by cultural composition of crew

Each governance model brings unique conflict resolution mechanisms:
- Hierarchical: Command decisions (fast, +15% speed, -10% satisfaction)
- Representative: Mediation panels (balanced, neutral satisfaction impact)
- Constitutional: Legal procedures (slow, -15% speed, +10% satisfaction)
- Cultural: Community consensus (variable speed, high satisfaction when successful)

4.6 Space-Born Religions and Philosophies
--------------------------------------
The game features emergent belief systems that develop in deep space over generations:

### The Void Observers
A contemplative tradition focused on the vastness of space as a spiritual teacher.
- Core Beliefs: The universe has consciousness; silence reveals truth; human scale is humbling
- Practices: "Void meditation" sessions; documentation of cosmic phenomena as spiritual texts
- Gameplay Impact: +15% science skill development; -10% social conflict chance; occasional "insight" events providing solution options

### Terrestrial Revivalism
A movement dedicated to preserving Earth culture with religious devotion.
- Core Beliefs: Earth as a sacred homeland; duty to preserve human heritage; eventual return
- Practices: Earth memorial days; archiving Earth data; maintaining "soil sanctuaries"
- Gameplay Impact: +20% hydroponics efficiency; resistance to certain adaptation technologies; strong social cohesion (+10% relationship building)

### Cosmic Pragmatism
A secular philosophy prioritizing survival through rational adaptation.
- Core Beliefs: Adaptation as highest virtue; elimination of non-functional traditions; evidence-based decision making
- Practices: Regular efficiency reviews; ceremonial adoption of innovations; "wasteful thinking" purging
- Gameplay Impact: +15% engineering efficiency; -5% morale during crises; accelerated technology development

4.7 Player Decision System for Cultural Direction
---------------------------------------------
Players influence cultural evolution through key decisions and policies:

- **Cultural Preservation vs. Innovation:**
  Choice between funding cultural preservation efforts (maintaining Earth traditions) or innovation initiatives (developing space-native cultures).
  Impact: Determines dominant value systems in later generations and relationship dynamics between generations.

- **Religious Policy Framework:**
  Establish ship-wide policies on religious observances, from secular (religion as private matter) to integrated (official recognition and accommodation).
  Impact: Affects morale, crew cohesion, and the emergence of new belief systems.

- **Educational Curriculum:**
  Shape what the next generation learns about Earth, cultural heritage, and identity.
  Impact: Directly influences the values and beliefs of future generations of crew members.

These cultural and religious systems create unique socio-cultural dynamics in each playthrough, adding depth to crew interactions and society building across generations.

5.0 Generational Mechanics
==========================
The Crew System models the evolution of crew members across multiple generations, accounting for genetic inheritance, environmental adaptation, and cultural drift.

5.1 Multi-Generational Crew Evolution
----------------------------------
The time-skip system models generational changes in crew values, culture, and governance based on multi-generational space society research:

### Cultural Divergence Mechanics
- Each 20-year time skip applies cultural drift effects
- Language: 5-7% vocabulary drift per 20 years, affecting cross-generational communication
- Values: Original mission objectives erode by 15-20% per generation
- Memory: Historical events lose significance at 30% per generation

### Generation Ship Governance
- Crew sizes <150: Direct democracy with all-crew voting (-5% efficiency, +20% morale)
- Crew sizes 150-1000: Council system with representative leaders (-10% efficiency, +10% morale)
- Crew sizes 1000+: Formalized governance with laws and elected officials (+10% efficiency, -5% morale)

```csharp
// Sample code for applying generational changes during time skip
public void ApplyGenerationalChanges(GameState state, int yearsSkipped)
{
    float culturalDriftFactor = (yearsSkipped / 20f) * CULTURAL_DRIFT_RATE;
    
    // Apply to all crew members
    foreach (CrewMember crew in state.GetAllCrewMembers())
    {
        // Apply drift only to those born on the mission
        if (crew.GenerationNumber > 0)
        {
            // Cultural values drift from original mission
            foreach (MissionValue value in crew.Values)
            {
                value.DriftFromOriginal(culturalDriftFactor * crew.GenerationNumber);
            }
            
            // Language drift affects communication
            if (crew.GenerationNumber >= 2)
            {
                // Apply language barriers with first-generation crew
                foreach (CrewMember otherCrew in state.GetCrewMembersByGeneration(0))
                {
                    float communicationPenalty = LANGUAGE_DRIFT_PENALTY * (crew.GenerationNumber - 1);
                    relationshipManager.ApplyCommunicationBarrier(crew, otherCrew, communicationPenalty);
                }
            }
            
            // Historical knowledge erosion
            foreach (HistoricalEvent historicalEvent in state.GetHistoricalEvents())
            {
                float significanceReduction = HISTORICAL_MEMORY_DECAY * crew.GenerationNumber;
                crew.ReduceHistoricalEventSignificance(historicalEvent, significanceReduction);
            }
        }
    }
    
    // Apply governance changes based on population
```

5.2 Generational Adaptation System
-------------------------------
The system tracks biological adaptation across generations based on space-born advantages:

```csharp
/// <summary>
/// Manages long-term biological adaptation across generations
/// </summary>
public class GenerationalAdaptationSystem
{
    // Tracks population-level adaptation metrics
    private Dictionary<AdaptationType, float> populationAdaptationLevels;
    
    // Calculate adaptation inheritance for a new crew member
    public void CalculateOffspringAdaptations(CrewMember offspring, CrewMember parentA, 
                                           CrewMember parentB)
    {
        foreach (AdaptationType adaptationType in System.Enum.GetValues(typeof(AdaptationType)))
        {
            float parentAValue = parentA.GetAdaptation(adaptationType);
            float parentBValue = parentB.GetAdaptation(adaptationType);
            
            // Base inheritance is average of parents plus genetic variation
            float baseInheritance = (parentAValue + parentBValue) / 2f;
            float variation = Random.Range(-GENETIC_VARIATION, GENETIC_VARIATION);
            
            // Apply environmental boost for space-born children
            float environmentalBoost = 0f;
            if (offspring.IsBornInSpace)
            {
                environmentalBoost = SPACE_BORN_ADAPTATION_BOOST;
            }
            
            // Calculate final adaptation value
            float adaptationValue = baseInheritance + variation + environmentalBoost;
            offspring.SetAdaptation(adaptationType, Mathf.Clamp01(adaptationValue));
        }
    }
}
```

5.3 Identity Evolution
-------------------
Colonial identity strengthens with each generation while Earth identity weakens:

```csharp
void UpdateColonistIdentities(Colony colony) {

    foreach (Colonist colonist in colony.GetAllColonists()) {

                colonialIdentity = 1f - earthIdentity;
                break;
                
            case 1: // Second generation
                earthIdentity = Mathf.Lerp(0.5f, 0.3f, divergence.values);
                colonialIdentity = 1f - earthIdentity;
                break;
                
            default: // Third+ generation
                earthIdentity = Mathf.Lerp(0.2f, 0.1f, divergence.values);
                colonialIdentity = 1f - earthIdentity;
                break;

        // Adjust based on personal factors
        if (colonist.HasTrait(TraitType.EarthNostalgia)) {
            earthIdentity += 0.15f;
            colonialIdentity -= 0.15f;
        }
        
        if (colonist.HasTrait(TraitType.Pioneer)) {
            earthIdentity -= 0.1f;
            colonialIdentity += 0.1f;
        }

        colonist.SetIdentityValues(earthIdentity, colonialIdentity);
```

6.0 Relationships and Social Networks

6.2 Relationship Evolution Algorithm
----------------------------------
Relationships evolve organically through a sophisticated algorithm that processes interactions, environmental factors, and personality compatibility:

```csharp
/// <summary>
/// Updates all active relationships based on interactions and environmental factors.
/// </summary>
/// <param name="deltaTime">Time elapsed since last update.</param>
public void UpdateRelationships(float deltaTime)
{
    // Get current environmental factors that affect all relationships
    EnvironmentalFactors currentFactors = environmentSystem.GetCurrentFactors();
    
    // Process all active relationships
    foreach (CrewRelationship relationship in activeRelationships)
    {
        // Apply time-based tension decay (relationships naturally cool down after conflicts)
        relationship.ApplyTensionDecay(deltaTime);
        
        // Apply environmental impacts (high stress, close quarters, etc.)
        relationship.ApplyEnvironmentalModifiers(currentFactors);
        
        // Process pending interactions for this relationship pair
        List<InteractionData> pendingInteractions = interactionQueue.GetPendingInteractions(
            relationship.CrewIdA, 
            relationship.CrewIdB
        );
        
        foreach (InteractionData interaction in pendingInteractions)
        {
            // Apply interaction effects to relationship
            relationship.UpdateFromInteraction(interaction);
            
            // Mark interaction as processed
            interactionQueue.MarkProcessed(interaction.Id);
            
            // Log significant relationship changes

            {
                relationshipHistory.LogSignificantInteraction(
                    relationship, 
                    interaction,
                    relationship.GetSnapshot()
                );
            }
        }
        
        // Check for relationship state transitions
        CheckForRelationshipStateTransitions(relationship);
    }
    
    // Update relationship network visualization
    relationshipVisualizer.UpdateNetwork(activeRelationships);
```

### Interaction Processing
Each interaction between crew members affects their relationship based on multiple factors:

```csharp
/// <summary>
/// Updates relationship values based on a specific interaction between crew members.
/// </summary>
/// <param name="interaction">Data about the interaction that occurred.</param>
public void UpdateFromInteraction(InteractionData interaction)
{
    // Calculate base impact factors from interaction type
    RelationshipImpactFactors baseImpact = relationshipImpactTable.GetBaseImpact(
        interaction.Type
    );
    
    // Get personality compatibility between the crew members
    float personalityCompatibility = personalityCompatibilityMatrix.GetCompatibility(
        crewManager.GetCrewMember(CrewIdA).PersonalityTraits,
        crewManager.GetCrewMember(CrewIdB).PersonalityTraits
    );
    
    // Calculate affinity change
    float affinityChange = baseImpact.AffinityImpact * 
                           personalityCompatibility *
                           interaction.Intensity;
    
    // Calculate trust impact - more significant from critical events
    float trustImpact = baseImpact.TrustImpact * 
                        interaction.Intensity *
                        (interaction.IsCritical ? CRITICAL_EVENT_MULTIPLIER : 1.0f);
    
    // Calculate respect impact - heavily influenced by skill demonstration
    float respectImpact = baseImpact.RespectImpact * 
                          interaction.SkillRelevance * 
                          interaction.Intensity;
    
    // Calculate familiarity impact - grows from any interaction but at different rates
    float familiarityImpact = baseImpact.FamiliarityImpact * 
                              (1.0f - (Familiarity / 100.0f)) * // Diminishing returns
                              interaction.Duration;
    
    // Apply changes with stability constraints to prevent wild swings
    Affinity = Mathf.Clamp(Affinity + affinityChange, -100.0f, 100.0f);
    Trust = Mathf.Clamp(Trust + trustImpact, 0.0f, 100.0f);
    Respect = Mathf.Clamp(Respect + respectImpact, 0.0f, 100.0f);
    Familiarity = Mathf.Clamp(Familiarity + familiarityImpact, 0.0f, 100.0f);
    
    // Add to shared experiences in appropriate category
    if (SharedExperiences.ContainsKey(interaction.Category))
    {
        SharedExperiences[interaction.Category] += interaction.Intensity;
    }
    else
    {
        SharedExperiences.Add(interaction.Category, interaction.Intensity);
    }
}
```

6.3 Environmental Impacts on Relationships
----------------------------------------
Environmental conditions strongly influence relationship development, especially in the confined spaces of space habitats:

```csharp
/// <summary>
/// Applies environmental modifiers to relationship values.
/// </summary>
/// <param name="factors">Current environmental conditions.</param>
public void ApplyEnvironmentalModifiers(EnvironmentalFactors factors)
{
    // Close quarters pressure increases familiarity but can also increase tension
    float quartersTightness = factors.LivingSpacePerCrew / OPTIMAL_SPACE_PER_CREW;
    if (quartersTightness < 1.0f)
    {
        // Close quarters increases familiarity
        float familiarityIncrease = (1.0f - quartersTightness) * 0.2f;
        Familiarity = Mathf.Clamp(Familiarity + familiarityIncrease, 0f, 100f);
        
        // But may also increase tension if compatibility is low
        if (Affinity < 0)
        {
            CurrentTension += (1.0f - quartersTightness) * 0.3f * (-Affinity/100f);
        }
    }
    
    // High stress situations can increase tension
    float stressTensionIncrease = factors.StressLevel * 0.1f;
    CurrentTension += stressTensionIncrease;
    
    // Social isolation can decrease familiarity and increase tension
    float isolationDecrease = factors.Isolation * 0.1f;
    CurrentTension += isolationDecrease;
    
    // Shared experiences can reduce tension
    float sharedExperienceReduction = SharedExperiences.Values.Sum() * 0.05f;
    CurrentTension = Mathf.Max(0, CurrentTension - sharedExperienceReduction);
}
```

```csharp
/// Checks if relationship values have crossed thresholds that would change relationship status.

public void CheckForRelationshipStateTransitions(CrewRelationship relationship)
{
    // Get current state
    RelationshipState currentState = relationshipStateManager.GetState(relationship);

        relationship.Affinity > ROMANTIC_THRESHOLD &&

            crewManager.GetCrewMember(relationship.CrewIdA),

            narrativeManager.CreateRomanticOpportunity(relationship, romanticCompatibility);

    if (currentState == RelationshipState.Neutral && 
        relationship.Affinity > FRIENDSHIP_THRESHOLD &&
        relationship.Familiarity > FRIENDSHIP_FAMILIARITY_THRESHOLD)

        narrativeManager.NotifyRelationshipChange(
```

6.5 Family Dynamics and Generational Relationships
-----------------------------------------------
The system models family relationships across generations, with unique dynamics between parents, children, and siblings:

### Parent-Child Relationships
- Inherit baseline affinity from parents' relationship quality
- Initial high trust and respect for parents, which may evolve during adolescence 
- Strong impact on child's personality and value development

### Sibling Relationships
- Influenced by birth order and age gap
- More likely to develop rivalries than other relationship types
- Shared childhood experiences create strong familial bonds even during conflicts

### Genetic and Trait Inheritance
Offspring inherit traits from parents with some variation:

```csharp
/// <summary>
/// Determines offspring traits based on parents' traits with some random variation.
/// </summary>
/// <param name="parentA">First parent.</param>
/// <param name="parentB">Second parent.</param>
/// <returns>Set of traits for the offspring.</returns>
public List<Trait> DetermineOffspringTraits(CrewMember parentA, CrewMember parentB)
{
    List<Trait> offspringTraits = new List<Trait>();
    
    // Get parent traits
    List<Trait> parentATraits = parentA.GetTraits();
    List<Trait> parentBTraits = parentB.GetTraits();
    
    // Traits that appear in both parents have high inheritance chance (80%)
    foreach (Trait trait in parentATraits)
    {
        if (parentBTraits.Contains(trait) && Random.value < 0.8f)
        {
            offspringTraits.Add(trait);
        }
    }
    
    // Other traits have lower inheritance chance (40%)
    foreach (Trait trait in parentATraits.Union(parentBTraits).Distinct())
    {
        if (!offspringTraits.Contains(trait) && Random.value < 0.4f)
        {
            offspringTraits.Add(trait);
        }
    }
    
    // Chance for a new emergent trait (30%)
    if (Random.value < 0.3f)
    {
        Trait newTrait = traitGenerator.GenerateEmergentTrait(parentATraits, parentBTraits);
        offspringTraits.Add(newTrait);
    }
    
    return offspringTraits;
}
```

6.6 Cultural Compatibility and Social Cohesion
-------------------------------------------
Cultural and religious identities strongly influence relationship development. The system calculates cultural compatibility between crew members:

```csharp
/// <summary>
/// Calculates cultural compatibility between two crew members.
/// </summary>
/// <param name="other">The other crew member's cultural identity.</param>
/// <returns>Compatibility score from -1.0 (incompatible) to 1.0 (highly compatible)</returns>
public float CalculateCompatibility(CulturalIdentity other)
{
    float compatibility = 0f;
    
    // Compare value systems (highest weight in compatibility)
    foreach (ValueAxis axis in ValueSystem.Keys)
    {
        if (other.ValueSystem.ContainsKey(axis))
        {
            float difference = Mathf.Abs(ValueSystem[axis] - other.ValueSystem[axis]);
            compatibility -= difference * VALUE_COMPATIBILITY_WEIGHT;
        }
    }
    
    // Compare belief systems
    if (PrimaryBelief == other.PrimaryBelief)
    {
        compatibility += SHARED_BELIEF_BONUS * 
                        (BeliefCommitment + other.BeliefCommitment) / 2f;
    }
    else if (BeliefDatabase.AreCompatible(PrimaryBelief, other.PrimaryBelief))
    {
        compatibility += COMPATIBLE_BELIEF_BONUS;
    }
    else if (BeliefDatabase.AreConflicting(PrimaryBelief, other.PrimaryBelief))
    {
        compatibility -= CONFLICTING_BELIEF_PENALTY * 
                       (BeliefCommitment + other.BeliefCommitment) / 2f;
    }
    
    // Shared rituals build connection
    foreach (RitualPractice ritual in Rituals)
    {
        if (other.Rituals.Contains(ritual))
        {
            compatibility += SHARED_RITUAL_BONUS;
        }
    }
    
    return Mathf.Clamp(compatibility, -1f, 1f);
}
```

### Cultural Policy Impacts on Social Cohesion
Ship-wide cultural policies affect overall social cohesion:

```csharp
/// <summary>
/// Calculates the impact of cultural policies on overall social cohesion.
/// </summary>
/// <param name="habitat">The habitat with crew members and cultural policies.</param>
/// <returns>Impact value affecting overall morale and community stability.</returns>
public float CalculateCohesionImpact(Habitat habitat)
{
    float impact = 0f;
    
    // Earth emphasis impact depends on generations from Earth
    float averageGeneration = habitat.Crew.Average(c => c.GenerationFromEarth);
    float optimalEarthEmphasis = Mathf.Max(0.9f - (averageGeneration * 0.15f), 0.3f);
    impact += 10f - (Math.Abs(EarthEmphasisLevel - optimalEarthEmphasis) * 20f);
    
    // Resource allocation directly impacts cohesion
    impact += (ResourceAllocation - 0.5f) * 25f;
    
    // Religious accommodation impact depends on religious population
    float religiousPercentage = habitat.Crew.Count(c => c.ReligiousDevoutness > 0.5f) / (float)habitat.Crew.Count;
    float optimalAccommodation = 0.3f + (religiousPercentage * 0.6f);
    impact += 10f - (Math.Abs(ReligiousAccommodation - optimalAccommodation) * 30f);
    
    // Expression freedom vs unity is complex
    // Too much freedom can reduce cohesion, too little causes resentment
    float expressionImpact = 0f;
    if (ExpressionFreedom < 0.3f)
        expressionImpact = (ExpressionFreedom - 0.3f) * 40f; // Severe impact if too restrictive
    else if (ExpressionFreedom > 0.8f)
        expressionImpact = (0.8f - ExpressionFreedom) * 25f; // Moderate impact if too free
    else
        expressionImpact = 5f; // Bonus if balanced
        
    impact += expressionImpact;
    
    return impact;
}
```
