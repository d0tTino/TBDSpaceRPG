```powershell

# Test menu item
./mcpWebSocketClient-simple.ps1 -menuPath "MCP/Ship/V2/Create Test Ship"

# Check server status
Test-NetConnection -ComputerName localhost -Port 8001 -InformationLevel Quiet
```
